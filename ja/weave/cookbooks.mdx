  </Card>

  <Card title="高度なトピック" icon="code-branch" href="/ja/weave/cookbooks/multi-agent-structured-output">

  <Card title="本番運用とモニタリング" icon="shield-check" href="/ja/weave/cookbooks/online_monitoring">
