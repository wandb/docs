```python
"""
Add one or more tags to previously saved runs.

Use the Public API to update tags after the run finishes or when you're
working outside the run process.

To add a new tag to a run, append it to the run's `tags` list, then call the
run's `update()` method to save the changes. The run's path consists of
`entity/project/run_id`.
"""
import wandb

entity = "[ENTITY]"
project = "[PROJECT]"
run_id = "[RUN-ID]"  # Replace with the ID of the run you want to update

# Path consists of entity/project/run_id
with wandb.Api().run(f"{entity}/{project}/{run_id}") as run:
  run.tags.append("[TAG]")
  run.update()
```
