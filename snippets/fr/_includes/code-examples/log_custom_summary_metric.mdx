"""
Enregistrer une métrique de résumé personnalisée dans W&B.
"""
import wandb
import random 

with wandb.init(project="<project>") as run:
    # Enregistrer une métrique de résumé personnalisée avec une valeur entière aléatoire entre 1 et 10
    run.summary["<metric_name>"] = random.randint(1, 10)
