export const MolabLink = ({ url, embed = false }) => {
  const embedSrc =
    url.replace(/^https:\/\/molab\.marimo\.io/, 'https://marimo.app') +
    '/wasm?embed=true';

  const button = (
    <a
      href={url}
      target="_blank"
      rel="noopener noreferrer"
      className="molab-link"
    >
      <img
        src="/images/marimo-icon.svg"
        alt=""
        width={16}
        height={16}
        className="molab-link__logo"
      />
      Run in molab
      <svg
        xmlns="http://www.w3.org/2000/svg"
        width="14"
        height="14"
        viewBox="0 0 24 24"
        fill="none"
        stroke="currentColor"
        strokeWidth="2"
        strokeLinecap="round"
        strokeLinejoin="round"
        aria-hidden="true"
        className="molab-link__external"
      >
        <path d="M7 17 17 7M17 7H9M17 7v8" />
      </svg>
    </a>
  );

  if (!embed) {
    return button;
  }

  return (
    <>
      <h2 className="cookbook-molab-heading">
        Try in marimo{' '}
        <span className="cookbook-try-badge">
          <Badge stroke shape="rounded" color="yellow" size="sm">
            New
          </Badge>
        </span>
      </h2>
      {button}
      <Frame>
        <iframe
          src={embedSrc}
          sandbox="allow-scripts allow-same-origin allow-downloads allow-popups allow-forms"
          allow="microphone"
          allowFullScreen
          loading="lazy"
          style={{
            width: '100%',
            height: '85vh',
            minHeight: '800px',
            border: 'none',
          }}
        />
      </Frame>
    </>
  );
};
