---
title: 로그 테이블
---

```python
"""
W&B에 테이블을 로그합니다.
"""
import wandb

# 두 개의 열과 두 개의 행 데이터로 테이블 객체 생성
my_table = wandb.Table(
    columns=["a", "b"],
    data=[["a1", "b1"], ["a2", "b2"]],
    log_mode="<log_mode>"
    )

# 새 run 시작
with wandb.init(project="<project>") as run:
    # W&B에 테이블 로그
    run.log({"<table_name>": my_table})
```
