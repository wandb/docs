```python
'''
W&B Weave의 자동 패칭을 사용하여 LLM에 대한 기본 요청을 트레이스합니다.
'''

# W&B 임포트하기
import weave
from openai import OpenAI

client = OpenAI()

# Weave가 지원하는 LLM에 요청을 보내는 함수 정의
def llm_request(question: str) -> dict:
    response = client.chat.completions.create(
    model="gpt-4o",
    messages=[
        {
            "role": "system",
            "content": """You are a helpful assistant."""
        },
        {
            "role": "user",
            "content": question
        }
        ],
        response_format={ "type": "text" }
    )
    return response.choices[0].message.content

# 팀 및 프로젝트 이름으로 Weave 초기화
weave.init('<your-team-name>/<your-project-name>')

question = "What is the capitol of Paris?"

# Weave 초기화 후 함수 호출
result = llm_request(question)

print(result)
```
